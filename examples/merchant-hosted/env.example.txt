# =============================================================================
# Merchant Hosted Example - Environment Configuration
# =============================================================================
# This example shows a merchant using a HOSTED facilitator
# =============================================================================

# Your merchant wallet address (where you receive USDC)
MERCHANT_ADDRESS=0xYourWalletAddress

# Facilitator URL (hosted by SDK provider)
# For testing locally: http://localhost:3001
# For production: https://your-facilitator-domain.com
FACILITATOR_URL=http://localhost:3001

# Server port
PORT=3000

# Network (base or base-sepolia)
NETWORK=base-sepolia

# =============================================================================
# SETUP INSTRUCTIONS
# =============================================================================
#
# 1. Rename this file from env.example.txt to .env
# 2. Replace MERCHANT_ADDRESS with YOUR wallet address
# 3. Update FACILITATOR_URL if using hosted service
# 4. Run: pnpm install
# 5. Run: pnpm dev
#
# You will receive USDC payments to MERCHANT_ADDRESS on Base!
#
